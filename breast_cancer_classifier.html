# We use StratifiedKFold to keep class proportions identical in every fold
# Extremely important for imbalanced datasets (Breast Cancer Dataset: 63 bening/ 37 malignant)
cv = StratifiedKFold(n_splits=CV_FOLDS, shuffle=True, random_state=RANDOM_STATE)
cv_results = []
count = 0

for name, model in pipelines.items():
    count+=1
    print(f"Evaluating {name}...")
    acc = cross_val_score(model, X_train, y_train, cv=cv, scoring='accuracy')
    roc = cross_val_score(model, X_train, y_train, cv=cv, scoring='roc_auc') # Preferred metric for medical diagnostics (False negatives are costly)
    print(count, name, acc)
    cv_results.append([name, acc.mean(), roc.mean()])
